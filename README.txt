#USAGE

		       [||:|_ENCRYPT_|:||]


FILENAME://:>  {The name of the text file that will be generated
		containing the message}

KEYNAME:///:>  {The name of the text file that will be generated
		containing the key}

MESSAGE:///:>  {Where the message to be encrypted is entered}


		       [||:|_DECRYPT_|:||]


FILE://:> {File name of the encrypted .txt}

KEY:///:> {File name of the key .txt}
